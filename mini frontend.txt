    <script>
        // Form submission handler
        document.getElementById('cropForm').addEventListener('submit', function(e) {
            e.preventDefault();
            
            const formData = {
                cropType: document.getElementById('cropType').value,
                location: document.getElementById('location').value,
                acreage: document.getElementById('acreage').value,
                plantingDate: document.getElementById('plantingDate').value
            };
            
            // Add new crop to the list
            addCropToList(formData);
            
            // Reset form
            this.reset();
            
            // Show success message
            alert('Crop data uploaded successfully! 🌱');
        });
        
        function addCropToList(cropData) {
            const cropList = document.getElementById('cropList');
            const newCrop = document.createElement('div');
            newCrop.className = 'crop-item';
            newCrop.style.animation = 'slideIn 0.5s ease';
            
            const plantingDate = new Date(cropData.plantingDate).toLocaleDateString();
            
            newCrop.innerHTML = `
                <div class="crop-info">
                    <div class="crop-name">${cropData.cropType.charAt(0).toUpperCase() + cropData.cropType.slice(1)} Field</div>
                    <div class="crop-details">Your Farm • ${cropData.acreage} acres • Planted ${plantingDate}</div>
                </div>
                <div class="crop-status">New</div>
            `;
            
            cropList.insertBefore(newCrop, cropList.firstChild);
        }
        
        // Map pin hover effects
        const mapPins = document.querySelectorAll('.map-pin');
        const tooltip = document.getElementById('tooltip');
        
        mapPins.forEach(pin => {
            pin.addEventListener('mouseenter', function(e) {
                const crop = this.getAttribute('data-crop');
                const farmer = this.getAttribute('data-farmer');
                tooltip.innerHTML = `<strong>${farmer}</strong><br>${crop}`;
                tooltip.style.opacity = '1';
            });
            
            pin.addEventListener('mousemove', function(e) {
                tooltip.style.left = e.pageX + 10 + 'px';
                tooltip.style.top = e.pageY - 10 + 'px';
            });
            
            pin.addEventListener('mouseleave', function() {
                tooltip.style.opacity = '0';
            });
        });
        
        // Alert toggle function
        function toggleAlert(button) {
            if (button.textContent === 'Active') {
                button.textContent = 'Paused';
                button.style.background = 'rgba(255, 255, 255, 0.6)';
                button.style.color = '#666';
            } else {
                button.textContent = 'Active';
                button.style.background = 'rgba(255, 255, 255, 0.3)';
                button.style.color = '#333';
            }
        }
        
        // Show alert setup (placeholder)
        function showAlertSetup() {
            alert('Alert setup feature coming soon! You\'ll be able to set custom alerts for specific crops, locations, and market conditions.');
        }
        
        // Simulate real-time updates
        setInterval(() => {
            // Update insights with slight variations
            const totalAcres = document.querySelector('.insight-number');
            const currentValue = parseInt(totalAcres.textContent);
            if (Math.random() > 0.7) {
                totalAcres.textContent = currentValue + Math.floor(Math.random() * 5);
            }
        }, 10000);
        
        // Add some interactivity to crop items
        document.addEventListener('click', function(e) {
            if (e.target.closest('.crop-item')) {
                const item = e.target.closest('.crop-item');
                item.style.transform = 'scale(0.98)';
                setTimeout(() => {
                    item.style.transform = 'scale(1)';
                }, 150);
            }
        });
    </script>
